# Types

# LoadAction

type LoadAction = {
  loaded(count?: number, pageSize?: number): void;
  noMore(): void;
  noResults(): void;
  error(): void;
};

Describes possible actions in loaded prop callback.

# LoadPayload

type LoadPayload = {
  isFirstLoad: boolean;
};
